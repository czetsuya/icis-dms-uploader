'link local tables
Public Function eLinkLocalTables() As Boolean
    Dim db As Database
    Dim strInput As String, strDB As String
    Set db = CurrentDb
    
    strDB = ""
    strDB = LaunchCD(Form_frmUploadDMS, "Please select the LOCAL DMS database.")

    If IsNull(strDB) Or Trim(strDB) = "" Then
       eLinkLocalTables = False
       Exit Function
    End If
    
    Dim tempTbl(0 To 17) As String
    tempTbl(0) = "STUDY"
    tempTbl(1) = "FACTOR"
    tempTbl(2) = "LEVEL_C"
    tempTbl(3) = "LEVEL_N"
    tempTbl(4) = "VARIATE"
    tempTbl(5) = "EFFECT"
    tempTbl(6) = "DMSATTR"
    tempTbl(7) = "DATA_C"
    tempTbl(8) = "DATA_N"
    tempTbl(9) = "OINDEX"
    tempTbl(10) = "VEFFECT"
    tempTbl(11) = "TRAIT"
    tempTbl(12) = "SCALE"
    tempTbl(13) = "TMETHOD"
    tempTbl(14) = "STEFFECT"
    tempTbl(15) = "LEVELS"
    tempTbl(16) = "OBSUNIT"
    tempTbl(17) = "REPRESTN"
    
    For Each tbl In tempTbl
        If Not FindTable(tbl) Then
            DoCmd.TransferDatabase acLink, "Microsoft Access", strDB, acTable, tbl, tbl
        End If
    Next
    eLinkLocalTables = True
End Function

End Sub
